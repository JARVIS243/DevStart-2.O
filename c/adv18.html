Create new names (aliases) for existing data types.


#include <stdio.h>

typedef unsigned int uint;
typedef struct {
    int id;
    char name[50];
} Student;

int main() {
    uint age = 25;
    Student s1 = {101, "Anu"};
    printf("Student: %s, Age: %u\n", s1.name, age);
    return 0;
}

ğŸ”¢ enum

Create named constants with integer values.


#include <stdio.h>

enum Day { MON, TUE, WED, THU, FRI, SAT, SUN };

int main() {
    enum Day today = WED;
    printf("Day number: %d\n", today); // 2
    return 0;
}
