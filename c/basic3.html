#include <stdio.h>

int main() {
    int age;
    char name[30];

    printf("Enter your age: ");
    scanf("%d", &age);

    printf("Enter your name: ");
    scanf("%s", name);

    printf("Hello %s, you are %d years old.\n", name, age);

    return 0;
}

Explanation:

  • scanf("%d", &age); reads an integer from user input.
  • scanf("%s", name); reads a string (word) without spaces.
  • Always use & before variable names for non-array inputs.
