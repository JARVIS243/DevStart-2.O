#include <stdio.h>

struct Student {
    int id;
    char name[50];
};

int main() {
    struct Student s = {1, "Anu"};
    FILE *fp = fopen("student.bin", "wb");
    fwrite(&s, sizeof(s), 1, fp);
    fclose(fp);
    return 0;
}

#include <stdio.h>

struct Student {
    int id;
    char name[50];
};

int main() {
    struct Student s;
    FILE *fp = fopen("student.bin", "rb");
    fread(&s, sizeof(s), 1, fp);
    printf("ID: %d, Name: %s\n", s.id, s.name);
    fclose(fp);
    return 0;
}
