🌳 Binary Tree Structure


struct Node {
    int data;
    struct Node *left, *right;
};

void inorder(struct Node* root) {
    if (root == NULL) return;
    inorder(root->left);
    printf("%d ", root->data);
    inorder(root->right);
}
