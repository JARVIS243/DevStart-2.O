Advanced C: Lesson 11 – Structures and Pointers

Access and manipulate structure data using pointers

👤 Basic Structure


struct Student {
    int id;
    char name[50];
    float grade;
};

🔗 Accessing Members with Pointers


#include <stdio.h>

struct Student {
    int id;
    char name[50];
    float grade;
};

int main() {
    struct Student s1 = {101, "Anu", 92.5};
    struct Student *ptr = &s1;

    printf("ID: %d\n", ptr->id);
    printf("Name: %s\n", ptr->name);
    printf("Grade: %.2f\n", ptr->grade);

    return 0;
}
