Advanced C: Lesson 14 – File Record Search

Search specific records from a file containing structures

📖 File with Multiple Records

Suppose you have a file with multiple Student structures stored.


struct Student {
    int id;
    char name[50];
};

void searchRecord(int key) {
    struct Student s;
    FILE *fp = fopen("student.bin", "rb");
    int found = 0;

    while (fread(&s, sizeof(s), 1, fp)) {
        if (s.id == key) {
            printf("Found: %d - %s\n", s.id, s.name);
            found = 1;
            break;
        }
    }

    if (!found) {
        printf("Record not found.\n");
    }

    fclose(fp);
}
