TypeScript – Advanced 5: Custom Type Guards

Safely narrow down types at runtime using custom logic

🔍 What is a Type Guard?

A type guard is a function that checks and narrows a variable's type.


function isString(value: unknown): value is string {
  return typeof value === "string";
}

function printLength(value: unknown) {
  if (isString(value)) {
    console.log("Length:", value.length);
  } else {
    console.log("Not a string.");
  }
}

🔎 Guarding Custom Types


type Dog = { bark: () => void };
type Cat = { meow: () => void };

function isDog(pet: Dog | Cat): pet is Dog {
  return (pet as Dog).bark !== undefined;
}

const myPet: Dog | Cat = { bark: () => console.log("Woof!") };

if (isDog(myPet)) {
  myPet.bark();
} else {
  myPet.meow();
}
