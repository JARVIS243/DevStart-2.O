Combine multiple types and use safe handling with unknown types

‚ûï Intersection Types

Use & to combine two or more types into one.


type Engineer = {
  name: string;
  skills: string[];
};

type Manager = {
  department: string;
};

type Lead = Engineer & Manager;

const teamLead: Lead = {
  name: "Aju",
  skills: ["TS", "React"],
  department: "Development"
};

‚ùî Unknown Type

unknown is safer than any because you must check the type before using it.


function handleValue(value: unknown) {
  if (typeof value === "string") {
    console.log("String:", value.toUpperCase());
  } else if (typeof value === "number") {
    console.log("Number squared:", value * value);
  } else {
    console.log("Unknown type");
  }
}

handleValue("hello");
handleValue(4);
handleValue(true);
