Interfaces can extend other interfaces to create complex hierarchies.


interface Animal {
  name: string;
}

interface Dog extends Animal {
  bark(): void;
}

const myDog: Dog = {
  name: "Tommy",
  bark() {
    console.log("Woof!");
  }
};

interface Flyer {
  fly(): void;

  swim(): void;

const duck: Duck = {
  fly() {
    console.log("Flying...");
  },
  swim() {
    console.log("Swimming...");
  }
};
