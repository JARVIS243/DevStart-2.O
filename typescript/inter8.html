Access deeply nested properties safely without crashing your code.


type User = {
  name: string;
  address?: {
    city?: string;
  };
};

const user1: User = {
  name: "Aju",
  address: {
    city: "Kochi"
  }
};

const user2: User = {
  name: "Nila"
};

console.log(user1.address?.city); // Kochi
console.log(user2.address?.city); // undefined (safe)
