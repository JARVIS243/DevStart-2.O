üîç What is Type Narrowing?

When you use conditional checks like typeof or in, TypeScript narrows down the possible types.


function printId(id: number | string) {
  if (typeof id === "string") {
    console.log("ID in lowercase:", id.toLowerCase());
  } else {
    console.log("ID doubled:", id * 2);
  }
}
printId("AJU"); // ID in lowercase: aju
printId(7);     // ID doubled: 14

type Admin = {
  name: string;
  role: string;
};

type User = {
  name: string;
  email: string;
};

function showDetails(person: Admin | User) {
  if ("role" in person) {
    console.log("Admin Role:", person.role);
  } else {
    console.log("User Email:", person.email);
  }
}
