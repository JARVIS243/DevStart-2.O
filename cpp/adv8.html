Concepts allow you to specify requirements for template parameters, improving code clarity and error messages.

#include <concepts>
#include <iostream>
using namespace std;

template <typename T>
concept Incrementable = requires(T x) {
    x++;
};

template <Incrementable T>
T increment(T value) {
    return value + 1;
}

int main() {
    cout << increment(5) << endl;
    // cout << increment("Hello") << endl; // Error: string not Incrementable

    return 0;
}
