#include <iostream>
#include <vector>
using namespace std;

int main() {
    vector<int> numbers;

    // Add elements
    numbers.push_back(10);
    numbers.push_back(20);
    numbers.push_back(30);

    // Access elements
    for (int i = 0; i < numbers.size(); i++) {
        cout << "Element " << i << ": " << numbers[i] << endl;
    }

    return 0;
}
