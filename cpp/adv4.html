Smart pointers are objects that manage the lifetime of dynamically allocated memory automatically.

#include <iostream>
#include <memory>
using namespace std;

class Demo {
public:
    Demo() { cout << "Constructor called\n"; }
    ~Demo() { cout << "Destructor called\n"; }
    void greet() { cout << "Hello from Demo\n"; }
};

int main() {
    unique_ptr<Demo> ptr = make_unique<Demo>();
    ptr->greet();

    // No need to delete ptr manually, destructor called automatically

    return 0;
}
