#include <iostream>
using namespace std;

int main() {
    int* ptr = new int;     // allocate memory for one int
    *ptr = 100;             // assign value
    cout << "Value: " << *ptr << endl;

    delete ptr;             // free allocated memory

    // Dynamic array
    int size = 3;
    int* arr = new int[size];
    for (int i = 0; i < size; i++) {
        arr[i] = i * 10;
    }
    for (int i = 0; i < size; i++) {
        cout << "arr[" << i << "] = " << arr[i] << endl;
    }

    delete[] arr;           // free array memory

    return 0;
}
