#include <iostream>
#include <thread>

using namespace std;

void printMessage(string msg) {
    cout << msg << endl;
}

int main() {
    thread t1(printMessage, "Hello from thread 1");
    thread t2(printMessage, "Hello from thread 2");

    // Wait for threads to finish
    t1.join();
    t2.join();

    return 0;
}
