#include <string>
using namespace std;

int main() {
    string name = "DevStart";
    cout << "Hello, " << name << "!" << endl;

    // Getting input
    string userName;
    cout << "Enter your name: ";
    getline(cin, userName);  // reads full line including spaces

    cout << "Welcome, " << userName << "!" << endl;

    return 0;
}
