A reference is an alias for another variable. It must be initialized when declared.

#include <iostream>
using namespace std;

int main() {
    int original = 10;
    int &ref = original; // ref is reference to original

    cout << "original = " << original << endl;
    cout << "ref = " << ref << endl;

    ref = 20; // changes original
    cout << "After modifying ref:" << endl;
    cout << "original = " << original << endl;
    cout << "ref = " << ref << endl;

    return 0;
}
