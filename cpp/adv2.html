#include <iostream>
#include <map>
using namespace std;

int main() {
    map<string, int> ageMap;

    // Insert key-value pairs
    ageMap["Aju"] = 21;
    ageMap["Deva"] = 22;
    ageMap["Rani"] = 20;

    // Access elements
    for (auto &entry : ageMap) {
        cout << entry.first << " is " << entry.second << " years old." << endl;
    }

    return 0;
}
