Inheritance allows a new class (derived) to inherit properties and behaviors from an existing class (base).

ğŸ§‘â€ğŸ’» Example: Base and Derived Classes


#include <iostream>
using namespace std;

class Animal {
public:
    void speak() {
        cout << "Animal speaks" << endl;
    }
};

class Dog : public Animal {
public:
    void bark() {
        cout << "Dog barks" << endl;
    }
};

int main() {
    Dog d;
    d.speak();  // inherited from Animal
    d.bark();   // own method

    return 0;
}
