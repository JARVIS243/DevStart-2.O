using System.IO;

class Program
{
    static void Main()
    {
        using (StreamReader reader = new StreamReader("data.txt"))
        {
            string content;
            while ((content = reader.ReadLine()) != null)
            {
                Console.WriteLine(content);
            }
        }
    }
}
