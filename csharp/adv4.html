ğŸ”” What is an Event?

An event is a way for a class to notify other classes when something happens. It uses delegates internally.

ğŸ“„ Event Example


using System;

class Alarm
{
    public delegate void AlarmHandler();
    public event AlarmHandler OnRing;

    public void Ring()
    {
        Console.WriteLine("Alarm is ringing...");
        OnRing?.Invoke();
    }
}

class Listener
{
    public void HandleAlarm()
    {
        Console.WriteLine("Listener: Alarm received!");
    }
}

// Usage
class Program
{
    static void Main()
    {
        Alarm alarm = new Alarm();
        Listener listener = new Listener();

        alarm.OnRing += listener.HandleAlarm;

        alarm.Ring();
    }
}
