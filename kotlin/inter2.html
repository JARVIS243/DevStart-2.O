ğŸ§± Secondary Constructor


class Book {
    var title: String
    var pages: Int

    constructor(title: String, pages: Int) {
        this.title = title
        this.pages = pages
    }

    fun info() {
        println("Book: $title, Pages: $pages")
    }
}

fun main() {
    val b1 = Book("DevStart Kotlin", 200)
    b1.info()
}
