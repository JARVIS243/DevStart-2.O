open class Animal {
    fun eat() {
        println("Eating...")
    }
}

class Dog : Animal() {
    fun bark() {
        println("Barking...")
    }
}

fun main() {
    val dog = Dog()
    dog.eat() // Inherited
    dog.bark()
}
