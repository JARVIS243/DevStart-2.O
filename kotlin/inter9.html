An interface defines methods but doesn't provide implementation (like a blueprint). Classes implement interfaces.


interface Drawable {
    fun draw()
}

class Circle : Drawable {
    override fun draw() {
        println("Drawing a circle")
    }
}

fun main() {
    val c = Circle()
    c.draw()
}
