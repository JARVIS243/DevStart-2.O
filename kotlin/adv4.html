Generics allow you to write code that works with different types without rewriting it.


class Box(val item: T) {
    fun get(): T = item
}

fun main() {
    val intBox = Box(123)
    val strBox = Box("DevStart")
    
    println(intBox.get())
    println(strBox.get())
}
