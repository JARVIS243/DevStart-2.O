Kotlin – Intermediate 8: Object Expression

Use anonymous objects for quick implementations

⚡ Anonymous Object


fun main() {
    val obj = object {
        val name = "DevStart"
        fun greet() = "Hello $name"
    }

    println(obj.greet())
}

🔄 Object Expression for Interfaces


interface Clickable {
    fun onClick()
}

fun main() {
    val btn = object : Clickable {
        override fun onClick() {
            println("Button clicked!")
        }
    }

    btn.onClick()
}
